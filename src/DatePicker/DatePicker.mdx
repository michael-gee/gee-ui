# Date Picker

A date picker component with range and presets.

---

The Date Picker is built using a composition of the `<Popover />` and the `<Calendar />` components.

Example usage:

```jsx
"use client"

import * as React from "react"
import { format } from "date-fns"
import { Calendar as CalendarIcon } from "lucide-react"

import { cn } from "@gee-ui/utils"
import { Button, Calendar, Popover, PopoverContent, PopoverTrigger } from "@gee-ui/react"

export function DatePickerDemo() {
  const [date, setDate] = React.useState<Date>()

  return (
    <Popover>
      <PopoverTrigger asChild>
        <Button
          variant={"outline"}
          className={cn(
            "w-[280px] justify-start text-left font-normal",
            !date && "text-muted-foreground"
          )}
        >
          <CalendarIcon className="mr-2 h-4 w-4" />
          {date ? format(date, "PPP") : <span>Pick a date</span>}
        </Button>
      </PopoverTrigger>
      <PopoverContent className="w-auto p-0">
        <Calendar
          mode="single"
          selected={date}
          onSelect={setDate}
          initialFocus
        />
      </PopoverContent>
    </Popover>
  )
}


```

For further details and examples, please refer to the shadcn documentation here: [shadcn/ui docs](https://ui.shadcn.com/docs/components/date-picker)
